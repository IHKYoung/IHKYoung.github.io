import os
import re
import glob
import shutil

"""
split_sceneflow.py
功能：
将SceneFlow数据集中的文件按照一定的规则复制到新的文件夹中
文件夹结构如下：
SceneFlowMerge
├── train
│   ├── disparity
│   ├── left
│   └── right
└── test
暂时未考虑test文件夹
"""
def generate_prefix_from_file_path(file_path):
    # 去掉文件名和文件扩展名
    descriptive_path = os.path.dirname(file_path).replace('./', '')
    # 移除 "frames_finalpass" 或 "disparity"
    descriptive_path = re.sub(r'(/frames_finalpass|/disparity|/left|/right)', '', descriptive_path)
    # 将路径中的 '/' 替换为 '_'
    return descriptive_path.replace('/', '_').replace('.', '').lower()

def copy_images_with_prefix_from_source_file(source_pattern, destination_folder, new_extension=None):
    source_files = glob.glob(source_pattern)
    
    if not os.path.exists(destination_folder):
        os.makedirs(destination_folder)

    for source_file in source_files:
        prefix = generate_prefix_from_file_path(source_file)
        filename = os.path.basename(source_file)
        filename_without_extension, extension = os.path.splitext(filename)

        if new_extension:
            extension = new_extension

        new_filename = f"{prefix}_{filename_without_extension}{extension}"
        destination_path = os.path.join(destination_folder, new_filename)

        if os.path.exists(destination_path):
            print(f"Skipping {source_file} because {destination_path} already exists")
            continue
        shutil.copy(source_file, destination_path)
        print(f"Copying {source_file} to {destination_path}")

# 定义所有需要遍历的路径模式
source_patterns = [
    './SceneFlow/Driving/frames_finalpass/*/*/*/left/*.png',
    './SceneFlow/FlyingThings3D/frames_finalpass/TRAIN/A/*/left/*.png',
    './SceneFlow/Monkaa/frames_finalpass/*/*/left/*.png'
]

if __name__ == "__main__":
    destination_folder_left_train = './SceneFlowMerge/train/left'
    destination_folder_right_train = './SceneFlowMerge/train/right'
    destination_folder_disp = './SceneFlowMerge/train/disparity'

    for source_pattern_left in source_patterns:
        source_pattern_right = source_pattern_left.replace('left', 'right')
        source_pattern_disp_left = source_pattern_left.replace('frames_finalpass', 'disparity').replace('.png', '.pfm')
        source_pattern_disp_right = source_pattern_right.replace('frames_finalpass', 'disparity').replace('.png', '.pfm')

        copy_images_with_prefix_from_source_file(source_pattern_left, destination_folder_left_train)
        copy_images_with_prefix_from_source_file(source_pattern_right, destination_folder_right_train)
        copy_images_with_prefix_from_source_file(source_pattern_disp_left, destination_folder_disp, new_extension='.pfm')
        copy_images_with_prefix_from_source_file(source_pattern_disp_right, destination_folder_disp, new_extension='.pfm')

import os

"""
check_sceneflow.py
功能：
验证test train disparity文件夹中的文件名是否一致
可以拓展使用
"""
def verify_file_names_without_extension(base_path):
    # 初始化保存不一致文件名的列表
    missing_in_right = []
    missing_in_disparity = []

    # 获取并排序'left'文件夹中的文件名
    left_folder = os.path.join(base_path, 'left')
    left_files = [os.path.splitext(f)[0] for f in sorted(os.listdir(left_folder))] if os.path.exists(left_folder) else []

    # 获取并排序'right'和'disparity'文件夹中的文件名
    right_folder = os.path.join(base_path, 'right')
    right_files = [os.path.splitext(f)[0] for f in sorted(os.listdir(right_folder))] if os.path.exists(right_folder) else []

    # disparity_folder = os.path.join(base_path, 'disparity')
    disparity_folder = os.path.join(base_path, 'dsp')
    disparity_files = [os.path.splitext(f)[0] for f in sorted(os.listdir(disparity_folder))] if os.path.exists(disparity_folder) else []

    for file_name in left_files:
        # 对于每个在'left'中找到的文件名，检查是否也在'right'和'disparity'中存在
        if file_name not in right_files:
            missing_in_right.append(file_name)

        if file_name not in disparity_files:
            missing_in_disparity.append(file_name)

    # 打印结果
    if not missing_in_right and not missing_in_disparity:
        print("All filenames are consistent across 'left', 'right', and 'disparity' folders.")
    else:
        if missing_in_right:
            for miss in missing_in_right:
                print(f"missing_in_right: {miss}")
        if missing_in_disparity:
            for miss in missing_in_disparity:
                print(f"missing_in_disparity: {miss}")
        print(f"length of missing_in_right: {len(missing_in_right)}")
        print(f"length of missing_in_disparity: {len(missing_in_disparity)}")

# 调用函数
# base_path = './SceneFlowMerge/train'
base_path = '/mnt/disk/DrivingStereo/train/'
verify_file_names_without_extension(base_path)

import numpy as np
import os
import re
import glob
import cv2

"""
disparity_pfm_to_png.py
功能：
将pfm格式的视差图转换为png格式
目前是转成256x的uint16格式
"""
# 读取PFM文件的函数
def load_pfm(file_path):
    with open(file_path, 'rb') as f:
        # 解析文件头
        header = f.readline().decode().rstrip()
        if header == 'PF':
            channels = 3
        elif header == 'Pf':
            channels = 1
        else:
            raise Exception("Not a PFM file.")

        # 获取图像尺寸
        dim_match = re.match(r'^(\d+)\s+(\d+)\s+$', f.readline().decode('utf-8'))
        if dim_match:
            width, height = map(int, dim_match.groups())
        else:
            raise Exception("Malformed PFM header.")

        # 读取数据
        scale = float(f.readline().decode().rstrip())
        data = np.fromfile(f, dtype=np.float32, count=channels * width * height)
        data = np.reshape(data, (height, width, channels))
        data = np.flipud(data)
        return data

# 读取PFM文件的函数（改进版）
def load_pfm_improved(file_path):
    with open(file_path, 'rb') as f:
        # 解析文件头，确保它是一个PFM文件
        header = f.readline().decode().rstrip()
        if header == 'PF':
            channels = 3
        elif header == 'Pf':
            channels = 1
        else:
            raise Exception("Not a PFM file.")

        # 获取图像尺寸
        dim_match = re.match(rb'^(\d+)\s(\d+)\s$', f.readline())
        if dim_match:
            width, height = map(int, dim_match.groups())
        else:
            raise Exception("Malformed PFM header.")

        # 获取并解析缩放因子，确定字节序
        scale = float(f.readline().decode().rstrip())
        if scale < 0:  # little-endian
            endian = '<'
            scale = -scale
        else:
            endian = '>'  # big-endian

        # 读取数据
        data = np.fromfile(f, endian + 'f', count=channels * width * height)

        # 根据通道数重塑数据
        shape = (height, width, channels) if channels > 1 else (height, width)
        data = np.reshape(data, shape)

        # 上下翻转数据以匹配PFM格式
        data = np.flipud(data)
        
        return data

# 主程序
def main():
    input_folder = './SceneFlowMerge/train/disparity'
    output_folder = './SceneFlowMerge/train/disparity_uint16'

    if not os.path.exists(output_folder):
        os.makedirs(output_folder)

    pfm_files = glob.glob(os.path.join(input_folder, '*.pfm'))

    for pfm_file in pfm_files:
        # 读取PFM文件
        disparity_data = load_pfm_improved(pfm_file)

        # 将浮点数乘以256并转换为uint16
        disparity_data_uint16 = np.uint16(disparity_data * 256)

        # 获取输出文件名和路径
        file_name = os.path.basename(pfm_file).replace('.pfm', '.png')
        output_file_path = os.path.join(output_folder, file_name)

        # 保存图像
        cv2.imwrite(output_file_path, disparity_data_uint16)

if __name__ == "__main__":
    main()
