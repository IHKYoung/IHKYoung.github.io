// 26. Remove Duplicates from Sorted Array
#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if (nums.size() <= 1) return nums.size();

        int slow = 1;
        for (int fast = 1; fast < nums.size(); fast++) {
            if (nums[fast - 1] != nums[fast]) {
                nums[slow++] = nums[fast];
            }
        }

        return slow;
        
    }
};
